-- ============================================================ -- Module 07 – Performance & Query Plans -- Exercise 1: EXPLAIN, Index Strategy, and Query Optimization -- ============================================================ -- SETUP: Run this section first to create and populate tables -- ============================================================ DROP TABLE IF EXISTS production_logs CASCADE; DROP TABLE IF EXISTS machines CASCADE; CREATE TABLE machines ( machine_id SERIAL PRIMARY KEY, machine_name VARCHAR(100) NOT NULL, factory_id INT NOT NULL, operational_status VARCHAR(50) NOT NULL ); CREATE TABLE production_logs ( log_id SERIAL PRIMARY KEY, machine_id INT NOT NULL REFERENCES machines (machine_id), product_code VARCHAR(50) NOT NULL, quantity INT NOT NULL, duration_seconds INT NOT NULL, log_timestamp TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ); INSERT INTO machines (machine_name, factory_id, operational_status) VALUES ('Mill-A1', 1, 'active'), ('Laser-B2', 1, 'active'), ('Robot-C1', 2, 'active'), ('Robot-C2', 2, 'maintenance'); INSERT INTO production_logs (machine_id, product_code, quantity, duration_seconds, log_timestamp) SELECT (ROW_NUMBER() % 4) + 1, 'PROD-' || LPAD(CAST(ROW_NUMBER() % 100 AS VARCHAR), 3, '0'), 100 + RANDOM() * 500, 300 + RANDOM() * 1000, CURRENT_TIMESTAMP - INTERVAL '1 day' * (ROW_NUMBER() / 100) FROM generate_series(1, 10000);

Challenge:

C1: Rewrite a slow query to avoid a full table scan. Provide before/after EXPLAIN plans.

📋 Solution 1

Q1 Solution:

EXPLAIN SELECT * FROM production_logs WHERE machine_id = 1; -- Expected output: Seq Scan (if no index) or Index Scan (if indexed) -- Plan structure: -- -> Seq Scan on production_logs -- Filter: (machine_id = 1) -- Explanation: Without an index, PostgreSQL performs a full table scan. -- Access Method: Sequential scan (Seq Scan)

Q2 Solution:

EXPLAIN ANALYZE SELECT m.machine_name, COUNT(pl.log_id) AS log_count FROM machines m LEFT JOIN production_logs pl ON m.machine_id = pl.machine_id WHERE m.factory_id = 1 GROUP BY m.machine_name; -- Output shows: -- Plan rows vs. actual rows -- Loop counts and actual timing -- Planning Time and Execution Time -- Helps identify where the planner made mistakes.

Q3 Solution:

CREATE INDEX idx_production_logs_machine_timestamp ON production_logs (machine_id, log_timestamp DESC); -- Reason for composite index: -- This index supports both the filter on machine_id and the range condition on log_timestamp. -- PostgreSQL can use the index for efficient row retrieval and avoid a full table scan. -- DESC on log_timestamp helps if you typically order by most recent first.

C1 Solution:

-- BEFORE: Slow query with full table scan EXPLAIN ANALYZE SELECT * FROM production_logs WHERE log_timestamp > CURRENT_TIMESTAMP - INTERVAL '7 days' AND quantity > 250; -- Expected: Seq Scan on production_logs (millions of rows scanned) -- AFTER: Optimized with index CREATE INDEX idx_production_logs_timestamp_qty ON production_logs (log_timestamp DESC, quantity); EXPLAIN ANALYZE SELECT * FROM production_logs WHERE log_timestamp > CURRENT_TIMESTAMP - INTERVAL '7 days' AND quantity > 250; -- Expected: Index Scan using idx_production_logs_timestamp_qty -- Far fewer rows scanned and returned immediately.
