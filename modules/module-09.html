Capstone

📖 Project Overview

You are building a production tracking database for a manufacturing facility. This capstone integrates:

  • Schema Design – CREATE tables with proper constraints
  • Data Integrity – FOREIGN KEYs, CHECK constraints
  • DML Operations – INSERT, UPDATE, DELETE with transactions
  • Advanced Queries – JOINs, CTEs, window functions
  • Automation – Triggers and procedures
  • Security – Roles and RLS policies
  • Performance – Indexes and query optimization

✏️ Capstone Project: Manufacturing Operations System

Capstone – Requirements & Deliverables

Scenario: A factory produces components on multiple machines. You must build:

Part 1: Schema Design

Create tables for:

  • Machines (machine_id, name, capacity_units_per_hour, status)
  • Work Orders (order_id, machine_id, product_code, target_quantity, deadline, status)
  • Production Runs (run_id, order_id, start_time, end_time, actual_quantity, quality_check_pass)
  • Downtime Events (event_id, machine_id, start_time, end_time, reason_code)
  • Quality Audit (audit_id, run_id, auditor_name, pass_fail, notes)

Part 2: Queries

  • Calculate machine efficiency % (actual vs. rated capacity)
  • Find overdue work orders
  • Rank machines by production volume in last 7 days
  • Identify quality trends per machine

Part 3: Automation

  • Create a trigger to mark orders as 'complete' when all runs pass QA
  • Create a procedure to log downtime and update machine status

Part 4: Security & Performance

  • Create roles: auditor_role (read-only on Quality Audit), operator_role (INSERT production runs)
  • Add index on frequently searched columns
  • Optimize the efficiency calculation query
📋 Complete Capstone Solution

Part 1: Schema Design

-- Clean start DROP TABLE IF EXISTS quality_audit CASCADE; DROP TABLE IF EXISTS downtime_events CASCADE; DROP TABLE IF EXISTS production_runs CASCADE; DROP TABLE IF EXISTS work_orders CASCADE; DROP TABLE IF EXISTS machines CASCADE; CREATE TABLE machines ( machine_id SERIAL PRIMARY KEY, machine_name VARCHAR(100) NOT NULL, capacity_units_per_hour INT NOT NULL CHECK (capacity_units_per_hour > 0), status VARCHAR(20) NOT NULL DEFAULT 'operational', created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ); CREATE TABLE work_orders ( order_id SERIAL PRIMARY KEY, machine_id INT NOT NULL REFERENCES machines (machine_id), product_code VARCHAR(50) NOT NULL, target_quantity INT NOT NULL CHECK (target_quantity > 0), deadline DATE NOT NULL, status VARCHAR(20) NOT NULL DEFAULT 'pending', created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ); CREATE TABLE production_runs ( run_id SERIAL PRIMARY KEY, order_id INT NOT NULL REFERENCES work_orders (order_id), start_time TIMESTAMP NOT NULL, end_time TIMESTAMP, actual_quantity INT NOT NULL CHECK (actual_quantity >= 0), quality_check_pass BOOLEAN NOT NULL DEFAULT TRUE ); CREATE TABLE downtime_events ( event_id SERIAL PRIMARY KEY, machine_id INT NOT NULL REFERENCES machines (machine_id), start_time TIMESTAMP NOT NULL, end_time TIMESTAMP, reason_code VARCHAR(50) NOT NULL ); CREATE TABLE quality_audit ( audit_id SERIAL PRIMARY KEY, run_id INT NOT NULL REFERENCES production_runs (run_id), auditor_name VARCHAR(100) NOT NULL, pass_fail VARCHAR(10) NOT NULL, notes TEXT, audit_date TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ); -- Indexes CREATE INDEX idx_work_orders_machine_deadline ON work_orders (machine_id, deadline); CREATE INDEX idx_production_runs_order ON production_runs (order_id); CREATE INDEX idx_downtime_machine_start ON downtime_events (machine_id, start_time); CREATE INDEX idx_quality_run ON quality_audit (run_id);

Part 2: Key Queries

-- Machine Efficiency % WITH machine_stats AS ( SELECT m.machine_id, m.machine_name, m.capacity_units_per_hour, SUM(pr.actual_quantity) AS total_produced, EXTRACT(EPOCH FROM (MAX(pr.end_time) - MIN(pr.start_time))) / 3600 AS total_hours FROM machines m LEFT JOIN work_orders wo ON m.machine_id = wo.machine_id LEFT JOIN production_runs pr ON wo.order_id = pr.order_id WHERE pr.start_time > CURRENT_TIMESTAMP - INTERVAL '7 days' GROUP BY m.machine_id, m.machine_name, m.capacity_units_per_hour ) SELECT machine_name, ROUND(100.0 * total_produced / NULLIF(capacity_units_per_hour * total_hours, 0), 2) AS efficiency_percent FROM machine_stats ORDER BY efficiency_percent DESC; -- Overdue Work Orders SELECT order_id, product_code, deadline, status FROM work_orders WHERE deadline < CURRENT_DATE AND status NOT IN ('complete', 'cancelled') ORDER BY deadline; -- Machine Production Ranking (Last 7 Days) SELECT m.machine_name, SUM(pr.actual_quantity) AS total_produced, ROW_NUMBER() OVER (ORDER BY SUM(pr.actual_quantity) DESC) AS rank FROM machines m JOIN work_orders wo ON m.machine_id = wo.machine_id JOIN production_runs pr ON wo.order_id = pr.order_id WHERE pr.start_time > CURRENT_TIMESTAMP - INTERVAL '7 days' GROUP BY m.machine_id, m.machine_name ORDER BY rank;

Part 3: Automation

-- Trigger: Mark orders complete when all runs pass QA CREATE OR REPLACE FUNCTION check_order_completion() RETURNS TRIGGER LANGUAGE plpgsql AS $$ DECLARE v_order_id INT; v_failed_count INT; BEGIN v_order_id := NEW.run_id; -- Count failed quality checks for this order SELECT COUNT(*) INTO v_failed_count FROM production_runs pr WHERE pr.order_id = (SELECT order_id FROM production_runs WHERE run_id = NEW.run_id) AND pr.quality_check_pass = FALSE; IF v_failed_count = 0 THEN UPDATE work_orders SET status = 'complete' WHERE order_id = (SELECT order_id FROM production_runs WHERE run_id = NEW.run_id); END IF; RETURN NEW; END; $$; CREATE TRIGGER trg_order_completion AFTER INSERT ON quality_audit FOR EACH ROW EXECUTE FUNCTION check_order_completion(); -- Procedure: Log downtime and update machine status CREATE OR REPLACE PROCEDURE log_downtime( p_machine_id INT, p_reason_code VARCHAR ) LANGUAGE plpgsql AS $$ BEGIN -- Insert downtime event INSERT INTO downtime_events (machine_id, start_time, reason_code) VALUES (p_machine_id, CURRENT_TIMESTAMP, p_reason_code); -- Update machine status UPDATE machines SET status = 'downtime' WHERE machine_id = p_machine_id; COMMIT; END; $$;

Part 4: Security & Roles

-- Create roles CREATE ROLE auditor_role NOLOGIN; CREATE ROLE operator_role NOLOGIN; -- Auditor: read-only access to quality_audit GRANT USAGE ON SCHEMA public TO auditor_role; GRANT SELECT ON quality_audit TO auditor_role; GRANT SELECT ON production_runs TO auditor_role; -- For context -- Operator: INSERT production runs, read work orders GRANT USAGE ON SCHEMA public TO operator_role; GRANT INSERT, SELECT ON production_runs TO operator_role; GRANT SELECT ON work_orders TO operator_role; GRANT SELECT ON machines TO operator_role;
