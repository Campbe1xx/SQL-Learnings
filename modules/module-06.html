📋 Solution 1

Q1 Solution:

CREATE VIEW production_summary AS SELECT m.machine_name, COUNT(pr.run_id) AS total_runs, SUM(pr.quantity_produced) AS total_quantity, ROUND(AVG(pr.duration_minutes), 2) AS avg_duration_minutes FROM machines m LEFT JOIN production_runs pr ON m.machine_id = pr.machine_id GROUP BY m.machine_id, m.machine_name ORDER BY m.machine_name;

Q2 Solution:

CREATE OR REPLACE PROCEDURE log_production_run( p_machine_id INT, p_product_name VARCHAR, p_quantity INT, p_duration INT, p_run_date DATE ) LANGUAGE plpgsql AS $$ DECLARE v_run_id INT; BEGIN INSERT INTO production_runs (machine_id, product_name, quantity_produced, duration_minutes, run_date) VALUES (p_machine_id, p_product_name, p_quantity, p_duration, p_run_date) RETURNING run_id INTO v_run_id; INSERT INTO audit_log (table_name, action_type, record_id) VALUES ('production_runs', 'INSERT', v_run_id); COMMIT; END; $$;

Q3 Solution:

CREATE OR REPLACE FUNCTION get_machine_efficiency(p_machine_id INT) RETURNS NUMERIC LANGUAGE plpgsql AS $$ DECLARE v_efficiency NUMERIC; BEGIN SELECT ROUND(AVG(quantity_produced::NUMERIC / NULLIF(duration_minutes, 0)), 2) INTO v_efficiency FROM production_runs WHERE machine_id = p_machine_id; RETURN COALESCE(v_efficiency, 0); END; $$;

Q4 Solution:

CREATE OR REPLACE FUNCTION audit_production_runs() RETURNS TRIGGER LANGUAGE plpgsql AS $$ BEGIN INSERT INTO audit_log (table_name, action_type, record_id) VALUES ('production_runs', TG_OP, NEW.run_id); RETURN NEW; END; $$; DROP TRIGGER IF EXISTS trg_audit_production_runs ON production_runs; CREATE TRIGGER trg_audit_production_runs AFTER INSERT OR UPDATE ON production_runs FOR EACH ROW EXECUTE FUNCTION audit_production_runs();

C1 Solution:

-- Setup for recursive CTE demo DROP TABLE IF EXISTS parts CASCADE; CREATE TABLE parts ( part_id SERIAL PRIMARY KEY, part_name VARCHAR(100), parent_part_id INT REFERENCES parts (part_id) ); INSERT INTO parts (part_id, part_name, parent_part_id) VALUES (1, 'Gearbox Assembly', NULL), (2, 'Primary Gear', 1), (3, 'Secondary Gear', 1), (4, 'Gear Tooth', 2), (5, 'Gear Shaft', 2); -- Recursive CTE WITH RECURSIVE part_hierarchy AS ( SELECT part_id, part_name, parent_part_id, 1 AS level, part_name AS hierarchy_path FROM parts WHERE parent_part_id IS NULL UNION ALL SELECT p.part_id, p.part_name, p.parent_part_id, ph.level + 1, ph.hierarchy_path || ' -> ' || p.part_name FROM parts p INNER JOIN part_hierarchy ph ON p.parent_part_id = ph.part_id ) SELECT part_id, part_name, hierarchy_path, level FROM part_hierarchy ORDER BY hierarchy_path;
